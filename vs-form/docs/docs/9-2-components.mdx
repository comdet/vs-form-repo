### Subschema

<ExtendLink link="ICommonProps" />

<ApiHeader>
  <ApiProp
  name="schemaName"
  type="string"
  description="Name of the subschema."
  />
  <ApiProp
  name="schema"
  type="object"
  description="The schema. Used internally"
  />
  <ApiProp
  name="data"
  type=<Link link="/docs-9-3-types/#ICommonDataProps" text="ICommonDataProps"/>
  />
  <ApiProp
  name="data.dataType"
  type="string (object or array)"
  description="If 'array', subschemas can be used and edited as arrays."
  />
  <ApiProp
  name="keyField"
  type="string"
  description="Provide a fieldname of the schema with unique values of each record (like an id). If not provided SchemaManager will add the keyfield '__rec_id__' with unique id's"
  />
  <ApiProp
  name="rowsPerPage"
  type="number"
  description="Defines number of rows per page."
  />
  <ApiProp
  name="hidePagination"
  type="boolean"
  description="If true, hides the pagination panel."
  />
  <ApiProp
  name="columnSettings"
  type="array of objects"
  description="Define the visible columns, if the subschema is an array and their width, etc. If not provided all fields will be shown with automatic width."
  />
  <ApiProp
  name="columnSettings.compId"
  type="string"
  description="The key of the component, which is shown."
  />
  <ApiProp
  name="columnSettings.autowidth"
  type="boolean"
  def="false"
  description="If true, width will be computed based on the content."
  />
  <ApiProp
  name="columnSettings.width"
  type="number"
  description="The width of the column. (If autowidth is false)"
  />
  <ApiProp
  name="columnSettings.widthUnit"
  def="px"
  type="string"
  description="The width unit. E.g. px, em, cm"
  />
  <ApiProp
  name="columnSettings.alignRight"
  def="false"
  type="boolean"
  description="Content is aligned right. Suitable for number values."
  />
  <ApiProp
  name="columnSettings.sortable"
  def="true"
  type="boolean"
  description="If true, the column is sortable."
  />
  <ApiProp
  name="columnSettings.sortDirection"
  def="none"
  type="string (none, asc or desc)"
  description="If sortable is true, defines the initial sortdirection."
  />

</ApiHeader>
